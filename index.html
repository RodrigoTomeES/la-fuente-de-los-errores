Subsetting fonts

@font-face {
 font-family: 'Awesome Font';
 font-style: normal;
 font-weight: 400;
 src: local('Awesome Font'),
      url('/fonts/awesome-l.woff2') format('woff2');
 /* Latin glyphs */
 unicode-range: U+000-5FF;
}

@font-face {
 font-family: 'Awesome Font';
 font-style: normal;
 font-weight: 400;
 src: local('Awesome Font'),
      url('/fonts/awesome-jp.woff2') format('woff2');
 /* Japanese glyphs */
 unicode-range: U+3000-9FFF, U+ff??;
}
